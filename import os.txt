import os
import openai
from dotenv import load_dotenv

load_dotenv()  # Loads .env file

openai.api_key = os.getenv("OPENAI_API_KEY")

response = openai.chat.completions.create(
    model="gpt-4",
    messages=[
        {"role": "system", "content": "You are a helpful assistant simulating a regular user in a sandboxed environment."},
        {"role": "user", "content": "Test the Liquid Glass overlay as a user would."}
    ]
)

print(response.choices[0].message.content)

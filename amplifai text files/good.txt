        // Panel Behavior
        const panel = document.getElementById('side-panel');
        panel.addEventListener('click', () => {
            panel.classList.toggle('locked');
        });
        
        // File Handling
        const fileInput = document.getElementById('file-input');
        document.querySelector('.btn-attach').addEventListener('click', () => fileInput.click());
        
        fileInput.addEventListener('change', (e) => {
            if(e.target.files[0].size > 100 * 1024 * 1024) {
                alert("File exceeds 100MB limit");
                return;
            }
            addMessage(`ðŸ“Ž ${e.target.files[0].name}`, true);
        });
        
        // Message Handling
        const chatHistory = document.getElementById('chat-history');
        const textarea = document.querySelector('textarea');
        const sendBtn = document.querySelector('.btn-send');
        
        function addMessage(text, isUser) {
            const msg = document.createElement('div');
            msg.className = isUser ? 'message user-message' : 'message bot-message';
            msg.textContent = text;
            chatHistory.appendChild(msg);
            chatHistory.scrollTop = chatHistory.scrollHeight;
        }
        
        // Send Message
        sendBtn.addEventListener('click', sendMessage);
        textarea.addEventListener('keydown', (e) => {
            if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault();
                sendMessage();
            }
        });
        
        function sendMessage() {
            const message = textarea.value.trim();
            if (message) {
                addMessage(message, true);
                textarea.value = '';
                
                // Simulate Amp typing
                setTimeout(() => {
                    addMessage("This is Amp's response. The interface now has:", false);
                    addMessage("- Paperclip in the exact spot as my UI", false);
                    addMessage("- Send button positioned correctly", false);
                    addMessage("- Settings button top-right", false);
                    addMessage("- Color splash effect", false);
                }, 800);
            }
        }
        
        // Settings Button
        document.getElementById('settings-btn').addEventListener('click', () => {
            alert("Settings panel would open here");
        });
